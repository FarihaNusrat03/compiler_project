#include<iostream>

int factorial(int n)
{
    int result;

    result = 1;

    for(int i = 1; i <= n; i = i + 1)
    {
        result = result * i;
    }

    return result;
}

int main()
{
    int x;
    bool flag;

    x = 5;
    flag = true;

    if(flag && x > 0)
    {
        cout<<x<<endl;
    }
    else
    {
        cout<<0<<endl;
    }

    while(x > 0)
    {
        x = x - 1;
    }

    return x;
}
